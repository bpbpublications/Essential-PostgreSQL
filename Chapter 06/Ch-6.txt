--Here, we are changing the deptno column name to dept_num of dept table:
alter table dept rename column deptno to dept_num


--Now rename column dnmae to dept_name:
alter table dept rename column dnmae to dept_name


--To change a default value of the column, you can use ALTER TABLE ALTER COLUMN SET DEFAULT or DROP DEFAULT:
ALTER TABLE table_name 
ALTER COLUMN column_name 
[SET DEFAULT value | DROP DEFAULT];

--To change the NOT NULL constraint, you can use ALTER TABLE ALTER COLUMN statement:
ALTER TABLE table_name 
ALTER COLUMN column_name 
[SET NOT NULL| DROP NOT NULL];
