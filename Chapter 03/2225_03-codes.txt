--Example of creating table: items
create table items(
item_id SERIAL primary key,
Item_name varchar(20) NOT NULL,

);

--Example of creating table: Books
create table Books(
Id SERIAL primary key,
title varchar(255),
attibute hstore
);

--To describe the table and check the length of columns, whether it is nullable or not
select column_name, data_type, is_nullable, character_maximum_length
from information_schema.columns
where table_name = 'books';
